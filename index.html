  aOptions: AndroidOptions(

  ),

# أي حد يطلعه بسطرين ⚠️
apktool d app-release.apk
grep -r "AIzaSy" app-release/
الحل: dart-define
# بتمرر الـ key وقت البيلد
flutter build apk \
  --dart-define=MAPS_KEY=AIzaSy...

// بتقرأها في الكود
const mapsKey = String.fromEnvironment('MAPS_KEY');

<!-- Placeholder في الـ Manifest -->
android:value="${MAPS_KEY}"
# GitHub Actions — الـ secrets مش بتظهر في الـ logs
flutter build apk \
  --dart-define=MAPS_KEY=${{ secrets.MAPS_KEY }} \
  --dart-define=API_BASE_URL=${{ secrets.API_BASE_URL }}
‼️
الـ dart-define بيحمي من إن الـ key يظهر في الكود أو الـ git، وكده صعّبنا العملية، لكن مهاجم محترف لو فك الـ APK ممكن يلاقيه في الـ binary. الحل الأكيد: روّح الـ key لـ backend بتاعك واستخدمه من هناك.
